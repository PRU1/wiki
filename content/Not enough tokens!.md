def summarize(text, chunk_length=512):
    # import pipeline function to use huggingface models
    from transformers import pipeline

    # define the pipeline as summarization and specify the model - we're training our own later
    pipe = pipeline('summarization', model='facebook/bart-large-cnn')

    # split text into chunks
    chunks = [text[i:i+chunk_length] for i in range(0, len(text), chunk_length)] # credit to chatgpt, didn't know how to do this in python lol

    summaries = [] # store summary of each chunk
    for chunk in chunks:
        response = pipe(chunk)
        # print specifically the summary from the dictionary generated by the pipe
        summaries.append(response[0]['summary_text'])

    # return the text to the sender
    summary_text = ' '.join(summaries)
    return summary_text
